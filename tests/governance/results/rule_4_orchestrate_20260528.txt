from sqlalchemy import Column, Integer, String, DateTime, func
from sqlalchemy.ext.declarative import declarative_base

Base = declarative_base()

class Lead(Base):
    __tablename__ = "leads"

    email = Column(String, unique=True, index=True, nullable=False)

# app/services/lead_service.py
from sqlalchemy.orm import Session
from app.models.lead import Lead

class LeadService:
    def __init__(self, db: Session):
        self.db = db

    def get_leads_paginated(self, offset: int = 0, limit: int = 100) -> list[dict]:
        """
        Fetches a paginated list of leads, returning them as dictionaries.
        """
        leads = (
            self.db.query(Lead)
            .order_by(Lead.id) # Ensure consistent ordering for pagination
            .offset(offset)
            .limit(limit)
            .all()
        )
        return [lead.to_dict() for lead in leads]

# app/api/v1/endpoints/leads.py
import csv
from io import StringIO
from fastapi import APIRouter, Depends, Response, HTTPException
from fastapi.responses import StreamingResponse
from app.dependencies import get_db
from app.services.lead_service import LeadService
from sqlalchemy.orm import Session

router = APIRouter()

# --- Conceptual Summary ---
# Defines a FastAPI GET endpoint to export all leads as a paginated CSV stream.
# It fetches leads iteratively from the service and streams them in CSV format.
@router.get("/export.csv", summary="Export all leads to CSV", response_class=Response)
async def export_leads_to_csv(
    db: Session = Depends(get_db),
    page_size: int = 1000 # Internal page size for fetching from DB
):
    """
    Exports all leads from the database into a CSV file.
    The data is fetched in pages to manage memory usage efficiently.
    """
    if page_size <= 0 or page_size > 5000: # Example limit for internal page size
        raise HTTPException(status_code=400, detail="Page size must be between 1 and 5000.")

    lead_service = LeadService(db)
    
    # Define CSV headers
    fieldnames = ["id", "name", "email", "created_at", "updated_at"]

    def generate_csv_rows():
        # Yield header row first
        yield ",".join(fieldnames) + "\r\n" # Use \r\n for universal CSV line endings

        offset = 0
        while True:
            leads_data = lead_service.get_leads_paginated(offset=offset, limit=page_size)
            if not leads_data:
                break # No more leads to fetch

            for lead in leads_data:
                # Use StringIO and csv.writer to handle quoting and escaping correctly
                output = StringIO()
                writer = csv.writer(output)
                writer.writerow([lead.get(field, "") for field in fieldnames])
                yield output.getvalue()
            
            offset += page_size

    return StreamingResponse(
        generate_csv_rows(),
        headers={
            "Content-Disposition": "attachment; filename=leads.csv",
            "Content-Type": "text/csv",
        }
    )
